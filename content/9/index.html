    Linien & Rechtecke zeichnen

    Zeichne im Paint‑Ereignis der Form einige Linien und ein gefülltes Rechteck.

    Lösung 
    private void Form1_Paint(object sender, PaintEventArgs e)
{
    var g = e.Graphics;
    g.DrawLine(Pens.Black, 10, 10, 200, 10);
    g.DrawRectangle(Pens.Blue, 20, 30, 120, 80);
    g.FillRectangle(Brushes.LightBlue, 22, 32, 116, 76);
}
    Kreis/Ellipse & Text

    Zeichne eine Ellipse und schreibe einen Text mittig darunter.

    Lösung 
    private void Form1_Paint(object sender, PaintEventArgs e)
{
    var g = e.Graphics;
    var rect = new Rectangle(50, 50, 150, 100);
    g.DrawEllipse(Pens.DarkGreen, rect);
    g.DrawString("Ellipse", this.Font, Brushes.Black, rect.Left, rect.Bottom + 5);
}
    Mauszeichnen – freihändig

    Implementiere Zeichnen per Maus: Beim Drücken beginnen, bei Bewegung Liniensegmente hinzufügen, beim Loslassen stoppen.

    Lösung 
    private bool drawing = false;
private Point last;

private void Form1_MouseDown(object sender, MouseEventArgs e) { drawing = true; last = e.Location; }
private void Form1_MouseUp(object sender, MouseEventArgs e) { drawing = false; }
private void Form1_MouseMove(object sender, MouseEventArgs e)
{
    if (!drawing) return;
    using var g = this.CreateGraphics();
    g.DrawLine(Pens.Black, last, e.Location);
    last = e.Location;
}

    Bild zeichnen

    Lade ein Bild aus Datei und zeichne es skaliert in die Form.

    Lösung 
    private Image? _img;

private void CmdLoadImg_Click(object sender, EventArgs e)
{
    using var dlg = new OpenFileDialog { Filter = "Bilder|*.png;*.jpg;*.jpeg;*.bmp" };
    if (dlg.ShowDialog() == DialogResult.OK) _img = Image.FromFile(dlg.FileName);
    Invalidate();
}

private void Form1_Paint(object sender, PaintEventArgs e)
{
    if (_img != null)
    {
        var g = e.Graphics;
        g.DrawImage(_img, new Rectangle(10,10,200,150));
    }
}
    Funktionsplot y = sin(x)

    Zeichne die Kurve y = sin(x) im Bereich x = 0 … 2π in ein Koordinatensystem.

    Lösung 
    private void Form1_Paint(object sender, PaintEventArgs e)
{
    var g = e.Graphics;
    int w = this.ClientSize.Width;
    int h = this.ClientSize.Height;
    PointF[] pts = new PointF[w];
    for (int x = 0; x < w; x++)
    {
        double t = (double)x / (w - 1) * 2 * Math.PI;
        double y = Math.Sin(t);
        float yy = (float)((1 - (y + 1) / 2) * (h - 1));
        pts[x] = new PointF(x, yy);
    }
    g.DrawLines(Pens.Red, pts);
}
    Persistentes Zeichnen – Liste von Formen

    Verwalte gezeichnete Formen in einer Liste und zeichne sie im Paint neu, statt sofort mit CreateGraphics.

    Lösung 
    private readonly List<Rectangle> rects = new();

private void CmdAddRect_Click(object s, EventArgs e)
{
    rects.Add(new Rectangle(10 + 20*rects.Count, 10, 40, 30));
    Invalidate();
}

private void Form1_Paint(object sender, PaintEventArgs e)
{
    foreach (var r in rects) e.Graphics.DrawRectangle(Pens.Black, r);
}
